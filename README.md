# mysite2026
## คำสั่งสร้าง project ทั้งหมด

    mkdir mysite2026
    cd mysite2026
    uv init
    uv add django
    uv run django-admin startproject mysite2026 .
    uv run manage.py startapp myapp
    

## คำสั่งที่ใช้
    uv run manage.py runserver [IP_ADDRESS]:80
